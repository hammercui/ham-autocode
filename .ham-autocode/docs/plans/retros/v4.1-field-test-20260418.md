  projectDir: string,

// auto-runner 在 batch 调度时收集 peer 的 declared files
const peerTasks = batch.map(b => readTask(projectDir, b.id)).filter(Boolean);
const peerFiles = peerTasks
  .filter(t => t.id !== task.id)
  .flatMap(t => t.files || []);

// Hashline verify 时 allowed = self + peers
